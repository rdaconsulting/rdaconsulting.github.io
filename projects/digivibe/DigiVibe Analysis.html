# Paired tests across multiple outcomes (tidyverse style) 
# ------------------------------------------------------------
paired_outcome_tests <- function(data,
                                 outcomes,
                                 id,
                                 group,
                                 group_levels = NULL,   # order: level2 - level1
                                 adjust = "BH",
                                 alternative = "two.sided",
                                 conf.level = 0.95,
                                 infer_reps = 0) {

  require(dplyr); require(tidyr); require(broom); require(purrr); require(rstatix)

  id_sym    <- rlang::ensym(id)
  group_sym <- rlang::ensym(group)

  df <- data |> dplyr::mutate(!!group_sym := as.factor(!!group_sym))

  if (is.null(group_levels)) {
    group_levels <- levels(dplyr::pull(df, !!group_sym))
    if (length(group_levels) != 2)
      stop("`group` must have exactly two levels or supply `group_levels`.")
  }

  df <- df |> dplyr::mutate(!!group_sym := forcats::fct_relevel(!!group_sym, group_levels))

  # ---- within-person differences: level2 - level1 ----
  pairs <- df |>
    tidyr::pivot_longer(tidyselect::all_of(outcomes), names_to = "outcome", values_to = "y") |>
    dplyr::group_by(outcome, !!id_sym) |>
    dplyr::filter(dplyr::n_distinct(!!group_sym) == 2) |>
    tidyr::pivot_wider(names_from = !!group_sym, values_from = y) |>
    dplyr::mutate(diff = .data[[as.character(group_levels[2])]] -
                        .data[[as.character(group_levels[1])]]) |>
    dplyr::select(outcome, !!id_sym, diff) |>
    dplyr::ungroup()

  # ---- diagnostics (n, shapiro, outliers, CL probs) ----
  diag_core <- pairs |>
    dplyr::group_by(outcome) |>
    dplyr::summarise(
      n_pairs   = sum(!is.na(diff)),
      n_pos     = sum(diff > 0,  na.rm = TRUE),
      n_neg     = sum(diff < 0,  na.rm = TRUE),
      n_tie     = sum(diff == 0, na.rm = TRUE),
      mean_diff = mean(diff, na.rm = TRUE),
      sd_diff   = sd(diff,   na.rm = TRUE),
      shapiro_p = if (n_pairs >= 3 && n_pairs <= 5000)
                    shapiro.test(diff)$p.value else NA_real_,
      .groups = "drop"
    )

  diag_outliers <- pairs |>
    dplyr::group_by(outcome) |>
    dplyr::group_modify(~{
      o <- rstatix::identify_outliers(.x, diff)
      tibble::tibble(
        n_outliers         = sum(o$is.outlier, na.rm = TRUE),
        n_outliers_extreme = sum(o$is.extreme, na.rm = TRUE)
      )
    }) |> dplyr::ungroup()

  diagnostics <- diag_core |>
    dplyr::left_join(diag_outliers, by = "outcome") |>
    dplyr::mutate(
      common_language_prob    = ifelse(n_pairs > 0,
                                       (n_pos + 0.5 * n_tie) / n_pairs, NA_real_),  # P(diff ≥ 0)
      common_language_prob_gt = dplyr::if_else(n_pos + n_neg > 0,
                                               n_pos / (n_pos + n_neg), NA_real_)   # P(diff > 0)
    )

  # ---- Cohen's d ----
  d_eff <- pairs |>
    dplyr::group_by(outcome) |>
    rstatix::cohens_d(diff ~ 1, mu = 0) |>
    dplyr::rename(cohens_d = effsize, d_magnitude = magnitude) |>
    dplyr::ungroup()

  # ---- paired t-test (one-sample on diffs) ----
  t_raw <- pairs |>
    dplyr::group_by(outcome) |>
    rstatix::t_test(diff ~ 1, mu = 0,
                    alternative = alternative,
                    conf.level  = conf.level,
                    detailed    = TRUE) |>
    dplyr::ungroup()

  if ("df" %in% names(t_raw)) {
    t_raw <- t_raw |> dplyr::rename(df_out = df)
  } else if ("parameter" %in% names(t_raw)) {
    t_raw <- t_raw |> dplyr::rename(df_out = parameter)
  } else {
    t_raw <- t_raw |> dplyr::mutate(df_out = NA_real_)
  }

  t_res <- t_raw |>
    dplyr::mutate(test = "paired_t") |>
    dplyr::left_join(diagnostics |> dplyr::select(outcome, n_pairs, mean_diff, sd_diff, shapiro_p,
                                                   n_outliers, n_outliers_extreme),
                     by = "outcome") |>
    dplyr::left_join(d_eff, by = "outcome") |>
    dplyr::mutate(p_adj = p.adjust(p, method = adjust)) |>
    dplyr::select(outcome, test,
                  estimate, conf.low, conf.high,
                  statistic, df = df_out, p, p_adj,
                  n_pairs, mean_diff, sd_diff, cohens_d, d_magnitude,
                  shapiro_p, n_outliers, n_outliers_extreme, method)

  # ---- Wilcoxon signed-rank + rank-biserial r + CL prob ----
  w_res <- pairs |>
    dplyr::group_by(outcome) |>
    dplyr::summarise(
      test_obj = list(stats::wilcox.test(diff, mu = 0,
                                         alternative = alternative,
                                         conf.int = TRUE,
                                         conf.level = conf.level,
                                         exact = FALSE)),
      n_eff = sum(!is.na(diff) & diff != 0),
      Tsum  = n_eff * (n_eff + 1) / 2,
      .groups = "drop"
    ) |>
    dplyr::mutate(tidy = purrr::map(test_obj, broom::tidy)) |>
    tidyr::unnest(tidy) |>
    dplyr::mutate(
      test = "wilcoxon_signed_rank",
      rank_biserial_r = dplyr::if_else(Tsum > 0, (2 * as.numeric(statistic) / Tsum) - 1, NA_real_),
      common_language_prob_approx = (rank_biserial_r + 1) / 2
    ) |>
    dplyr::rename(p = p.value) |>
    dplyr::mutate(p_adj = p.adjust(p, method = adjust)) |>
    dplyr::left_join(
      diagnostics |>
        dplyr::mutate(n_eff_calc = n_pos + n_neg) |>
        dplyr::select(outcome, n_pairs, n_eff_calc, n_pos, n_neg, n_tie,
                      common_language_prob, common_language_prob_gt),
      by = "outcome"
    ) |>
    dplyr::select(outcome, test,
                  estimate, conf.low, conf.high,
                  statistic, p, p_adj,
                  n_pairs, n_eff, n_pos, n_neg, n_tie,
                  rank_biserial_r,
                  common_language_prob,          # P(diff ≥ 0), ties = 0.5
                  common_language_prob_gt,       # P(diff > 0), ignore ties
                  common_language_prob_approx,   # from r_rb
                  method)

  out <- list(
    pairs = pairs,
    diagnostics = diagnostics,
    t_test = t_res,
    wilcoxon = w_res
  )

  # ---- Optional infer bootstrap for t-stat ----
  if (infer_reps > 0) {
    require(infer)
    dir_map <- c("two.sided" = "two_sided", "less" = "left", "greater" = "right")
    direction <- unname(if (!is.null(dir_map[[alternative]])) dir_map[[alternative]] else "two_sided")

    infer_t <- pairs |>
      dplyr::group_by(outcome) |>
      dplyr::group_modify(~{
        x <- .x$diff
        n <- sum(!is.na(x))
        m <- mean(x, na.rm = TRUE); s <- sd(x, na.rm = TRUE)
        t_obs <- (m - 0) / (s / sqrt(n))

        null_dist <- tibble::tibble(diff = x) |>
          infer::specify(response = diff) |>
          infer::hypothesize(null = "point", mu = 0) |>
          infer::generate(reps = infer_reps, type = "bootstrap") |>
          infer::calculate(stat = "t")

        tibble::tibble(t_stat_obs = t_obs,
                       p_infer = infer::get_p_value(null_dist, obs_stat = t_obs, direction = direction))
      }) |> dplyr::ungroup()

    out$t_test <- dplyr::bind_cols(out$t_test, infer_t |> dplyr::select(t_stat_obs, p_infer))
    out$infer_meta <- list(reps = infer_reps, direction = direction)
  }

  out
}

p_fmt <- function(x) ifelse(is.na(x), "", scales::pvalue(x, accuracy = 0.001, add_p = TRUE))
stars <- function(p) dplyr::case_when(
  is.na(p)   ~ "",
  p < 0.001  ~ "***",
  p < 0.01   ~ "**",
  p < 0.05   ~ "*",
  TRUE       ~ ""
)

make_ttest_gt <- function(res) {
  tab <- res$t_test |>
    transmute(
      outcome,
      estimate_ci = sprintf("%.3f (%.3f, %.3f)", estimate, conf.low, conf.high),
      t_df        = sprintf("%.2f (%s)", statistic, format(df, trim = TRUE)),
      p           = p,              # keep numeric for coloring/formatting
      p_adj       = p_adj,          # keep numeric
      sig         = stars(p_adj),
      cohens_d    = cohens_d,
      d_mag       = d_magnitude,
      n_pairs,
      shapiro_p   = shapiro_p,
      outliers    = n_outliers,
      extreme     = n_outliers_extreme
    )

  gt(tab, rowname_col = "outcome") |>
    cols_label(
      estimate_ci = html("Mean Δ (95% CI)"),
      t_df        = html("t (df)"),
      p           = html("<i>p</i>"),
      p_adj       = html("<i>p</i><sub>BH</sub>"),
      sig         = "sig",
      cohens_d    = "Cohen’s d",
      d_mag       = "magnitude",
      n_pairs     = "n (pairs)",
      shapiro_p   = html("Shapiro <i>p</i>"),
      outliers    = "outliers",
      extreme     = "extreme"
    ) |>
    # color by adjusted p-value (numeric column), set a sensible domain
    data_color(
      columns = p_adj,
      method  = "numeric",
      palette = c("#E8F5E9", "#FFFDE7", "#FFEBEE"), # green → yellow → red
      domain  = c(0, 0.15)                           
    ) |>
    # format numeric columns
    fmt(columns = c(p, p_adj, shapiro_p), fns = p_fmt) |>
    fmt_number(columns = cohens_d, decimals = 2) |>
    tab_style(
      style = cell_text(weight = "bold"),
      locations = cells_column_labels(everything())
    ) |>
    tab_spanner(label = "Estimate (95% CI)", columns = estimate_ci) |>
    tab_spanner(label = "Test", columns = c(t_df, p, p_adj, sig)) |>
    tab_spanner(label = "Effect size", columns = c(cohens_d, d_mag)) |>
    tab_spanner(label = "Diagnostics", columns = c(n_pairs, shapiro_p, outliers, extreme)) |>
    tab_footnote(
      footnote = "sig: significance based on BH-adjusted p-value (* < .05, ** < .01, *** < .001)."
    ) |>
    tab_options(table.font.size = px(13))
}


make_wilcox_gt <- function(res) {
  tab <- res$wilcoxon |>
    transmute(
      outcome,
      estimate_ci = sprintf("%.3f (%.3f, %.3f)", estimate, conf.low, conf.high),
      stat        = statistic,
      p           = p,            # numeric
      p_adj       = p_adj,        # numeric
      sig         = stars(p_adj),
      r_rb        = rank_biserial_r,
      clp_ties    = common_language_prob,    # P(Δ ≥ 0) ties=.5
      clp_gt      = common_language_prob_gt, # P(Δ > 0)
      n_pairs, n_eff, n_pos, n_neg, n_tie
    )

  gt(tab, rowname_col = "outcome") |>
    cols_label(
      estimate_ci = html("HL shift (95% CI)"),
      stat        = "W (R+)",
      p           = html("<i>p</i>"),
      p_adj       = html("<i>p</i><sub>BH</sub>"),
      sig         = "sig",
      r_rb        = "rank-biserial r",
      clp_ties    = "CL prob (ties=.5)",
      clp_gt      = "CL prob (>0)",
      n_pairs     = "n pairs", n_eff = "n eff", n_pos = ">0", n_neg = "<0", n_tie = "ties"
    ) |>
    data_color(
      columns = p_adj,
      method  = "numeric",
      palette = c("#E8F5E9", "#FFFDE7", "#FFEBEE"),
      domain  = c(0, 0.15)
    ) |>
    fmt(columns = c(p, p_adj), fns = p_fmt) |>
    fmt_number(columns = c(r_rb, clp_ties, clp_gt), decimals = 2) |>
    tab_style(
      style = cell_text(weight = "bold"),
      locations = cells_column_labels(everything())
    ) |>
    tab_spanner(label = "Estimate (95% CI)", columns = estimate_ci) |>
    tab_spanner(label = "Test", columns = c(stat, p, p_adj, sig)) |>
    tab_spanner(label = "Effect size", columns = c(r_rb, clp_ties, clp_gt)) |>
    tab_spanner(label = "Counts", columns = c(n_pairs, n_eff, n_pos, n_neg, n_tie)) |>
    tab_footnote(
      footnote = "CL prob ≈ probability a random participant’s dg=1 ≥ dg=0 (ties counted as 0.5)."
    ) |>
    tab_options(table.font.size = px(13))
}

Shapiro-Wilkes p-values and QQ plots suggest that pain scores appear to be normally distributed for deltoid injections, but not for fingerstick and abdomen injections . Thus, paired t-test is appropriate for the deltoid comparison, but Wilcoxon signed-rank test may be preferable for the others. It may be worthwhile to consider reporting both for all 3 outcomes, given the low sample size.

between_ttest_gt <- function(tt) {
  tab <- tt |>
    dplyr::transmute(
      analysis_label,
      grp1 = sprintf("%s (n=%d)  %.2f ± %.2f", group1, n1, m1, sd1),
      grp2 = sprintf("%s (n=%d)  %.2f ± %.2f", group2, n2, m2, sd2),
      estimate_ci = sprintf("%.2f (% .2f, % .2f)", estimate, conf.low, conf.high),
      t_df        = sprintf("%.2f (%s)", statistic, format(df, trim = TRUE)),
      p           = p, p_adj = p_adj, sig = sig_stars(p_adj),
      cohens_d    = cohens_d, magnitude
    )

mk_by_dg <- function(level_chr) {
  df <- dplyr::filter(dat, dg == level_chr)
  # ensure both gauge levels present
  if (dplyr::n_distinct(df$deltoid.gauge) < 2L) {
    empty_t <- tibble::tibble(
      analysis_label = character(), group1 = character(), group2 = character(),
      n1 = integer(), n2 = integer(), m1 = numeric(), sd1 = numeric(),
      m2 = numeric(), sd2 = numeric(), estimate = numeric(),
      conf.low = numeric(), conf.high = numeric(),
      statistic = numeric(), df = numeric(), p = numeric(),
      p_adj = numeric(), cohens_d = numeric(), magnitude = character()
    )
    empty_w <- tibble::tibble(
      analysis_label = character(), group1 = character(), group2 = character(),
      n1 = integer(), n2 = integer(), med1 = numeric(), med2 = numeric(),
      estimate = numeric(), conf.low = numeric(), conf.high = numeric(),
      statistic = numeric(), p = numeric(), p_adj = numeric(),
      rank_biserial_r = numeric(), cl_prob = numeric()
    )
    return(list(t = empty_t, w = empty_w))
  }

  sumB <- df |> dplyr::group_by(deltoid.gauge) |>
    dplyr::summarise(
      n   = dplyr::n(),
      m   = mean(deltoid),
      sd  = sd(deltoid),
      med = stats::median(deltoid),
      .groups = "drop"
    )

  g1 <- levels(df$deltoid.gauge)[1]
  g2 <- levels(df$deltoid.gauge)[2]
  s1 <- dplyr::filter(sumB, deltoid.gauge == g1)
  s2 <- dplyr::filter(sumB, deltoid.gauge == g2)

  ttB <- rstatix::t_test(deltoid ~ deltoid.gauge, var.equal = FALSE, detailed = TRUE, data = df)
  dB  <- rstatix::cohens_d(deltoid ~ deltoid.gauge, var.equal = FALSE, data = df)
  wxB <- broom::tidy(stats::wilcox.test(deltoid ~ deltoid.gauge, exact = FALSE, conf.int = TRUE, data = df))
  effB <- rstatix::wilcox_effsize(deltoid ~ deltoid.gauge, data = df)

  # df column robustly
  df_colB <- if ("df" %in% names(ttB)) ttB$df else if ("parameter" %in% names(ttB)) ttB$parameter else NA_real_

  U <- as.numeric(wxB$statistic) - s1$n * (s1$n + 1) / 2
  CL <- U / (s1$n * s2$n)
  label <- paste0("Deltoid (dg=", level_chr, ")")

  tt_row <- ttB |> dplyr::mutate(
    analysis_label = label,
    df = df_colB,
    p_adj = p,
    group1 = g1, group2 = g2,
    n1 = s1$n, n2 = s2$n,
    m1 = s1$m, sd1 = s1$sd,
    m2 = s2$m, sd2 = s2$sd,
    cohens_d = dB$effsize[1], magnitude = dB$magnitude[1]
  ) |>
    dplyr::select(analysis_label, group1, group2, n1, n2, m1, sd1, m2, sd2,
                  estimate, conf.low, conf.high, statistic, df, p, p_adj, cohens_d, magnitude)

  wx_row <- wxB |> dplyr::transmute(
    analysis_label = label,
    estimate, conf.low, conf.high,
    statistic = statistic, p = p.value, p_adj = p,
    group1 = g1, group2 = g2,
    n1 = s1$n, n2 = s2$n,
    med1 = s1$med, med2 = s2$med,
    rank_biserial_r = effB$effsize[1],
    cl_prob = CL
  ) |>
    dplyr::select(analysis_label, group1, group2, n1, n2, med1, med2,
                  estimate, conf.low, conf.high, statistic, p, p_adj, rank_biserial_r, cl_prob)

  list(t = tt_row, w = wx_row)
}
